fn funcion (x: i64, y: i64){
	let sum = x + y;
	let res = x - y;
	let mult = x * y;
	let div = x / y;
	println!(println!("{} + {} = {}",x,y,sum);
	println!(println!("{} + {} = {}",x,y,res);
	println!(println!("{} + {} = {}",x,y,mult);
	println!(println!("{} + {} = {}",x,y,div);
}

fn main(){
	let mut x:i64 = 5;
	let mut y:i64 = 10;

	funcion(x,y);
}
